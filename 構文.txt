<LED>
    led on r g b //任意の色のLEDをつける
        //r:0~10 g:0~10 b:0~10
    led time r g b t //任意の色のLEDを任意の時間つける
        //r:0~10 g:0~10 b:0~10 t:0.1~25.5
    led off //LEDをけす
    led in r g b t //任意の色のLEDをだんだん任意の時間つける(前回の色からだんだん変化する)
        //r:0~10 g:0~10 b:0~10 t:0.1~25.5
    led out r g b t //任意の色のLEDを任意の時間つけ、その後だんだん消えていく
        //r:0~10 g:0~10 b:0~10 t:0.1~25.5
<Sound>
    beep n //任意の番号の音を鳴らす
    beep play a //trueなら連続で音をならす、falseなら音をならす
        //a:true,false
    beep stop //音を止める
<Wait>
    wait t //任意の時間待つ
        //t:0.1~300
<Counter>
    count start //秒カウンタをスタートする
    count stop //秒カウンタを止める
    count reset //秒カウンタをリセットする
<If>
    if button o //ボタンがoならcを実行
        c
        //o:on,off c:各命令
    if light s n //明るさがnと比べてsなら
        c
        //s:<or>or== n:0~100or変数 c:各命令
    if sound o //音がするなら
        c
        //o:on,off c:各命令
    if alarm o //アラーム入力があるなら
        c
        //o:on,off c:各命令
    if timer o //タイマー入力があるなら
        c
        //o:on,off c:各命令
    if time s n1 n2 //時刻がn1時n2分と比べてsなら
        c
        //s:<or>or== n1:0~23 n2:0~59 c:各命令
    if temp s n //音温度がnと比べてsなら
        c
        //s:<or>or== n:-128~127or変数 c:各命令
    if random //50%の確率で実行
        c
        //c:各命令
    if count s n //秒カウンタがnと比べてなら
        c
        //s:<or>or== n:0~255 c:各命令
    if v s n //vがnと比べてsなら
        c
        //v:a~h s:<or>or== n:-128~127or変数 c:各命令
<Else>
    else //直前のifがfalseならcを実行
        c
        //c:各命令
<Variable>
    variable v s n
        //v:a~h s:+=or-=or= n:-128~127or変数
<For>
    for //ずっと繰り返す
    for n //n回繰り返す。
        //n:数値
<Until>
    until n button o //ボタンがoまでcをn回繰り返し(nがないならずっと)
        c
        //o:on,off c:各命令
    until n light s n //明るさがnと比べてsまでcをn回繰り返し(nがないならずっと)
        c
        //s:<or>or== n:0~100or変数 c:各命令
    until n sound o //音がoまでcをn回繰り返し(nがないならずっと)
        c
        //o:on,off c:各命令
    until n alarm o //アラーム入力があるまでcをn回繰り返し(nがないならずっと)
        c
        //o:on,off c:各命令
    until n timer o //タイマー入力があるまでcをn回繰り返し(nがないならずっと)
        c
        //o:on,off c:各命令
    until n time s n1 n2 //時刻がn1時n2分と比べてsまでcをn回繰り返し(nがないならずっと)
        c
        //s:<or>or== n1:0~23 n2:0~59 c:各命令
    until n temp s n //音温度がnと比べてsまでcをn回繰り返し(nがないならずっと)
        c
        //s:<or>or== n:-128~127or変数 c:各命令
    until n count s n //秒カウンタがnと比べてまでcをn回繰り返し(nがないならずっと)
        c
        //s:<or>or== n:0~255 c:各命令
    until n v s n //vがnと比べてsまでcをn回繰り返し(nがないならずっと)
        c
        //v:a~h s:<or>or== n:-128~127or変数 c:各命令
<disp>
    disp time //画面に時刻を表示する
    disp temp //画面に温度を表示する
    disp count //画面に秒カウンタを表示する
    disp light //画面に明るさを表示する
    disp wait //画面にウェイト時間を表示する
    disp v //画面に変数を表示する
        //v:a~h
    disp off //画面を消す
<sub>
    sub n //サブルーチンnを実行する
        //n:1~4


引数一覧
r 赤色の量
g 緑色の量
b 青色の量
t 時間
n 数字
a 真偽
c 命令
s 符号
v 変数名
